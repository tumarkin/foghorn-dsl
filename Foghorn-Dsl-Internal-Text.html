apostrophize :: Text -> Text #

Wrap text in single quotes.

>>> apostrophize "hello"
"'hello'"

parenthesize :: Text -> Text #

Wrap text in parentheses.

>>> parenthesize "a + b"
"(a + b)"

toSnakeCase :: Text -> Text #

Convert text to snake_case by inserting underscores before uppercase letters.

Each uppercase letter is converted to lowercase with a preceding underscore. Leading and trailing underscores are stripped from the result. Used to convert Haskell identifiers to SQL/database naming conventions.

>>> toSnakeCase "myVariableName"
"my_variable_name"

>>> toSnakeCase "PascalCase"
"pascal_case"

>>> toSnakeCase "HTTPResponse"
"h_t_t_p_response"

>>> toSnakeCase "_LeadingUnderscore"
"leading_underscore"

>>> toSnakeCase "already_snake_case"
"already_snake_case"
